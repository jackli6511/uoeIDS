data("NHANES")
data("NHANES", package = "NHANES")
write.csv(NHANES, "data/NHANES.csv", row.names = FALSE)

health <- read.csv("data/NHANES.csv")

library(dplyr)
library(ggplot2)
library(tidymodels)
library(scales) 

data_r <- health %>% select(
  ID,
  DaysPhysHlthBad,
  DaysMentHlthBad,
  SleepHrsNight,
  SleepTrouble
)

data_cl <- data_r %>% drop_na()

data_cl <- data_cl %>%
  mutate(
    HealthBad = DaysPhysHlthBad + DaysMentHlthBad
  ) %>%
  mutate(
    SleepCat = case_when(
      SleepHrsNight < 6 ~ "Short",
      SleepHrsNight >= 6 & SleepHrsNight <= 8 ~ "Normal",
      SleepHrsNight > 8 ~ "Long"
    )
  ) %>%
  mutate(
    SleepTrouble = factor(SleepTrouble, levels = c("No", "Yes"),
                          labels = c("No Trouble", "Trouble"))
  )

data_cl <- data_cl %>%
  mutate(
    HealthStatus = if_else(HealthBad > 0, "PoorHealth", "GoodHealth"),
    HealthStatus = factor(HealthStatus, levels = c("GoodHealth", "PoorHealth"))
  )

data_end <- data_cl %>%
  select(
    ID,
    HealthBad,
    HealthStatus,
    SleepHrsNight,
    SleepTrouble,
    SleepCat
  )
p_bar <- ggplot(data_end, aes(x = SleepCat, fill = HealthStatus)) +
  geom_bar(position = "fill") +
  scale_y_continuous(labels = percent) +
  labs(
    title = "Proportion of Poor Health by Sleep Duration Category",
    x = "Sleep Duration Category",
    y = "Proportion",
    fill = "Health Status"
  ) +
  theme_minimal()

p_bar

set.seed(123)

data_split <- initial_split(data_end, prop = 0.7, strata = HealthStatus)
train_data <- training(data_split)
test_data  <- testing(data_split)
health_rec <- recipe(HealthStatus ~ SleepHrsNight + SleepTrouble + SleepCat,
                     data = train_data) %>%
  step_dummy(all_nominal_predictors())
health_spec <- logistic_reg(mode = "classification") %>%
  set_engine("glm")
health_wf <- workflow() %>%
  add_recipe(health_rec) %>%
  add_model(health_spec)

health_fit <- fit(health_wf, data = train_data)
test_preds <- predict(health_fit, test_data, type = "prob") %>%
  bind_cols(test_data)
roc_df <- roc_curve(test_preds,
                    truth = HealthStatus,
                    .pred_PoorHealth)

p_roc <- autoplot(roc_df) +
  labs(
    title = "ROC Curve for HealthStatus Model",
    x = "1 - Specificity",
    y = "Sensitivity"
  )

p_roc

roc_auc(test_preds,
        truth = HealthStatus,
        .pred_PoorHealth)
